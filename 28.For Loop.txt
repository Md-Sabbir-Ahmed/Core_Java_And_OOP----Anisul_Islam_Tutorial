/* printing bangladesh */



package beginnerjava;


public class ForLoop {
    
    public static void main(String[] args) {
        
        
        for (int i = 1; i <= 100; i++) 
        
        {
           
        
        
        System.out.println(i+" Bangladesh");
        
        
        }
         
        
    }
    
    
}










/* even number printing */


package beginnerjava;


public class ForLoop {
    
    public static void main(String[] args) {
        
        
        for (int i = 2; i <= 100; i=i+2) 
        
        {
           
        
        
        System.out.println(i);
        
        
        }
         
        
    }
    
    
}





/* printing in descending order */






package beginnerjava;


public class ForLoop {
    
    public static void main(String[] args) {
        
        
        for (int i = 10 ; i>=1 ; i--) 
        
        {
           
        
        
        System.out.println(i);
        
        
        }
         
        
    }
    
    
}
