/* printing name with number */



package beginnerjava;


public class WhileLoop {
    
    
    public static void main(String[] args) {
        
        
        int i=1;
        
        while(i<=100)
        {
            
            System.out.println(i+" SABBIR");
            i++;
            
            
        }
        
        
        
        
    }
    
    
}
